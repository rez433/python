def add_to_char_fails_dict(char_fails_dict, char, mistakes_counter):
	mistakes_counter["fails"] += 1
	if char not in char_fails_dict:

	else:
		char_fails_dict[char] += 1

show_top_scores function

funktionen kommer att först skapa en tum lista "unsorted_scores". Sedan öppnar den och läser "score.txt" filen och sparar varje i en lista osh sedan läggar till varje lista till unsorted_scores.

Sedan med hjälp av lambda kommer den sortera listan efter score vilket är den andra varjabeln i varje list.

Sedan kommer den filtera listan efter level vilket är den tredje varjabeln i varje list. 


def show_top_scores():
	"""Show top scores"""
	unsorted_scores = []

	with open('./src/score.txt', 'r') as f:
		for line in f:
			line = line.strip()
			if not line:
				continue
			username, score, level = line.split()
			try:
				score = float(score)
			except ValueError:
				continue
			unsorted_scores.append((username, score, level))

	sorted_scores = sorted(unsorted_scores, key=lambda x: x[1], reverse=True)

	easy_scores = []
	medium_scores = []
	hard_scores = []

	for username, score, level in sorted_scores:
		if level == "easy":
			easy_scores.append((username, score))
		elif level == "medium":
			medium_scores.append((username, score))
		elif level == "hard":
			hard_scores.append((username, score))
